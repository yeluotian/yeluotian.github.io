def quick_sort(li):
    if len(li) <= 1:
	return li
    base = 0
    left = [val for val in li[base+1:] if val <= li[base]]
    right = [val for val in li[base+1:] if val > li[base]]
    return quick_sort(left) + [li[base]] + quick_sort(right)

# 归并排序
def merge_sort(li):
    if len(li) <= 1:
	return li
    
    mid = len(li) // 2
    left_list = merge_sort[:mid]
    right_list = merge_sort[mid:]
    left = right = 0
    ret = []
    while left < len(left_list) and right < len(right_list):
	if left_list[left] < right_list[right]:
	    ret.append(left_list[left])
	    left += 1
	else:
	    ret.append(right_list[right])
	    right += 1
    ret += left_list[left:]
    ret += right_list[right:]
    return ret

class SingleLinkList:
    def __init__(self, node=None):
        self.node = node
        
    def creat_link_list(self, li):
        p = self.node
        for val in li:
            p.next = Node(val)
            p = p.next
            
    def show_link_list(self):
        p = self.node
        p = p.next
        while p != None:
            print(p.val, end='-->')
            p = p.next
    
    def reverse_link_list(self):
        p = self.node
	pre = None
        while p != None:
            nextnode = p.next
            p.next = pre
	    pre = p
            p = nextnode
        self.show_link_list()

Reverse String 反转字符串，空间复杂度为o(1)

class Solution:
    def reverse_string_while(self, s: List(str)):
	start, end = 0, len(s) 
	while start < end:
	    s[start], s[end] = s[end], s[start]
	    start, end = start + 1, end -1
	return s

    def reverse_string_for(self, s: List(str)):
        lens = len(s)
        for i in range(lens // 2):
            s[i], s[lens-1-i] = s[lens-1-i], s[i]
        return s

Odd Even Linked List

class Solution:
    def oddEvenList(self, head: ListNode) -> ListNode:
        if head is None:
            return head
        pre, p = head, head.next
        phead = p
        
        while p != None and p.next != None:
            pre.next = p.next
            pre = pre.next
            
            p.next = pre.next
            p = p.next
            
        pre.next = phead
        return head

invert-binary-tree (20191024)

class Solution:
    def invertTree(self, root: TreeNode) -> TreeNode:
        if not root:
            return
        if root:
            root.left, root.right = root.right, root.left

            self.invertTree(root.left)
            self.invertTree(root.right)
        return root

is-subsequence (20191029)

class Solution:
    def isSubsequence(self, s: str, t: str) -> bool:
        lens = len(s)
        lent = len(t)
        idx_s, idx_t, cout = 0, 0, 0
        while idx_s < lens and idx_t < lent:
            if s[idx_s] == t[idx_t]:
                idx_s, idx_t = idx_s + 1, idx_t + 1
                cout += 1
            else:
                idx_t += 1
        if cout == lens:
            return True
        return False

class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        cur =0
        p = ListNode(0)
        l3 = p
        while l1 or l2 or cur:
            v1 = (l1.val if l1 else 0)
            v2 = (l2.val if l2 else 0)            
            cur, mod = divmod(v1+v2+cur, 10)
            l3.next = ListNode(mod)
            l3 = l3.next
            l1 = (l1.next if l1 else None)
            l2 = (l2.next if l2 else None)            
        return p.next
