905. Sort Array By Parity

  • Given an array A of non-negative integers, return an array consisting of all the even elements of A, followed by all the odd elements of A.
  • You may return any answer array that satisfies this condition.

思路：奇数排在偶数后面，用两个列表，分别存储偶数和奇数

class Solution:
    def sortArrayByParity(self, A: List[int]) -> List[int]:
        lena = len(A)
        left, right = [], []
        if lena <= 1:
            return A
        for val in A:
            if val % 2 == 0:
                left.append(val)
            else:
                right.append(val)
        ret = left + right
        return ret

78. Subsets

  • Given a set of distinct integers, nums, return all possible subsets (the power set).
  • Note: The solution set must not contain duplicate subsets.

思路：目标是求一个列表的子集，长度n的子集为2**n，即每个数都有两个选择，可将十进制转换为二进制，二进制对应位为1，则添加到子集

class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        lenn = len(nums)
        twolist = [[] for i in range(2 ** lenn)]
        for i in range(2 ** lenn):
            bins = self.dec2bin(i)
            if len(bins) < lenn:
                bins = '0' * (lenn-len(bins)) + bins
            for k,v in enumerate(bins):
                if v == '1':
                    twolist[i].append(nums[k])
        return twolist
        
    def dec2bin(self, n):
        div, mod = divmod(n, 2)
        s = str(mod)
        while div:
            div, mod = divmod(div, 2)
            s = str(mod) + s      
        return s

Reverse String 反转字符串，空间复杂度为o(1)

class Solution:
    def reverse_string_while(self, s: List(str)):
	start, end = 0, len(s) 
	while start < end:
	    s[start], s[end] = s[end], s[start]
	    start, end = start + 1, end -1
	return s

    def reverse_string_for(self, s: List(str)):
        lens = len(s)
        for i in range(lens // 2):
            s[i], s[lens-1-i] = s[lens-1-i], s[i]
        return s

Odd Even Linked List

class Solution:
    def oddEvenList(self, head: ListNode) -> ListNode:
        if head is None:
            return head
        pre, p = head, head.next
        phead = p
        
        while p != None and p.next != None:
            pre.next = p.next
            pre = pre.next
            
            p.next = pre.next
            p = p.next
            
        pre.next = phead
        return head

invert-binary-tree (20191024)

class Solution:
    def invertTree(self, root: TreeNode) -> TreeNode:
        if not root:
            return
        if root:
            root.left, root.right = root.right, root.left

            self.invertTree(root.left)
            self.invertTree(root.right)
        return root

is-subsequence (20191029)

class Solution:
    def isSubsequence(self, s: str, t: str) -> bool:
        lens = len(s)
        lent = len(t)
        idx_s, idx_t, cout = 0, 0, 0
        while idx_s < lens and idx_t < lent:
            if s[idx_s] == t[idx_t]:
                idx_s, idx_t = idx_s + 1, idx_t + 1
                cout += 1
            else:
                idx_t += 1
        if cout == lens:
            return True
        return False

add-two-numbers (20191030)

class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        cur =0
        p = ListNode(0)
        l3 = p
        while l1 or l2 or cur:
            v1 = (l1.val if l1 else 0)
            v2 = (l2.val if l2 else 0)            
            cur, mod = divmod(v1+v2+cur, 10)
            l3.next = ListNode(mod)
            l3 = l3.next
            l1 = (l1.next if l1 else None)
            l2 = (l2.next if l2 else None)            
        return p.next

rotate-list (20191031)

class Solution:
    def rotateRight(self, head: ListNode, k: int) -> ListNode:
        if not head:
            return
        p = head
        cout = 0
        while k:
            cur = p
            pre = p 
            p = p.next
            if not p:
                return pre
            while p.next:
                p = p.next
                pre = pre.next
                cout += 1
            p.next = cur
            pre.next = None
            
            div, mod = divmod(k, cout+2)
            if mod == 0:
                k = cout + 2
            else:
                k = mod
            k -= 1
        
        return p

  1. Image Overlap(20191113) Two images A and B are given, represented as binary, square matrices of the same size. (A binary matrix has only 0s and 1s as values.) 思路：图像经过翻转后，求最大重合部分，图像中的每一个点到另一个点的距离，距离相同最大的数，即为最大重合部分

    Input: A = [[1,1,0],
        [0,1,0],
        [0,1,0]]
   B = [[0,0,0],
        [0,1,1],
        [0,0,1]]
Output: 3

class Solution:
def largestOverlap(self, A: List[List[int]], B: List[List[int]]) -> int:
    lena = len(A)
    apos = [(row, col) for row in range(lena) for col in range(lena) if A[row][col]] 
    bpos = [(row, col) for row in range(lena) for col in range(lena) if B[row][col]]
    dist = {}  
    maxlap = 0
    for a in apos:
        for b in bpos:
            temp = (a[0]-b[0], a[1]-b[1])
            dist[temp] = dist.get(temp, 0) + 1 
            maxlap = max(maxlap, dist[temp]) 
    return maxlap

class Solution:
def mergeTwoLists(self, l1: ListNode, l2: ListNode) -> ListNode:
    if l1 == None and l2 == None:
        return None
    if l1 == None:
        return l2
    if l2 == None:
        return l1
        
    root = ListNode(0)
    p = root
    while l1 and l2:
        if l1.val <= l2.val:
            p.next = l1
            l1 = l1.next
        else:
            p.next = l2
            l2 = l2.next
        p = p.next
    if l1:
        p.next = l1
    else:
        p.next = l2
    return root.next

class Solution:
def mergeTwoLists(self, l1: ListNode, l2: ListNode) -> ListNode:
    if l1==None and l2==None:
        return None
    if l1==None:
        return l2
    if l2==None:
        return l1
    if l1.val<=l2.val:
        l1.next=self.mergeTwoLists(l1.next,l2)
        return l1
    else:
        l2.next=self.mergeTwoLists(l1,l2.next)
        return l2
